<script>
    const audioInput = document.getElementById('audioInput');
    const player = document.getElementById('player');
    const speedSlider = document.getElementById('speedSlider');
    const speedVal = document.getElementById('speedVal');
    const recordBtn = document.getElementById('recordBtn');
    const studioControls = document.getElementById('studioControls');
    const fileName = document.getElementById('fileName');

    let audioCtx, source, mediaRecorder, chunks = [];

    audioInput.onchange = (e) => {
        const file = e.target.files[0];
        if (file) {
            fileName.innerText = "Fichier chargé : " + file.name;
            player.src = URL.createObjectURL(file);
            studioControls.style.display = 'block';
            initStudio();
        }
    };

    function initStudio() {
        if (audioCtx) return;
        audioCtx = new (window.AudioContext || window.webkitAudioContext)();
        source = audioCtx.createMediaElementSource(player);

        // --- TRAITEMENT STUDIO PRO ---
        const equalizer = audioCtx.createBiquadFilter();
        equalizer.type = "peaking";
        equalizer.frequency.value = 2500; 
        equalizer.gain.value = 10; // Donne de la présence à la voix

        const compressor = audioCtx.createDynamicsCompressor();
        compressor.threshold.setValueAtTime(-24, audioCtx.currentTime);
        compressor.ratio.setValueAtTime(12, audioCtx.currentTime);

        const limiter = audioCtx.createDynamicsCompressor();
        limiter.threshold.setValueAtTime(-0.5, audioCtx.currentTime);

        source.connect(equalizer);
        equalizer.connect(compressor);
        compressor.connect(limiter);
        limiter.connect(audioCtx.destination);
    }

    speedSlider.oninput = () => {
        player.playbackRate = speedSlider.value;
        speedVal.innerText = speedSlider.value;
    };

    recordBtn.onclick = () => {
        if (recordBtn.innerText === "LANCER LE RENDU FINAL") {
            const stream = player.captureStream ? player.captureStream() : player.mozCaptureStream();
            mediaRecorder = new MediaRecorder(stream);
            chunks = [];

            mediaRecorder.ondataavailable = (e) => chunks.push(e.data);
            mediaRecorder.onstop = () => {
                const blob = new Blob(chunks, { type: 'audio/mp3' });
                const url = URL.createObjectURL(blob);
                const a = document.createElement('a');
                a.style.display = 'none';
                a.href = url;
                a.download = "Master_Studio_Express.mp3";
                document.body.appendChild(a);
                a.click();
                window.URL.revokeObjectURL(url);
                recordBtn.innerText = "LANCER LE RENDU FINAL";
                recordBtn.style.background = "linear-gradient(135deg, #00ff88 0%, #00b362 100%)";
            };

            mediaRecorder.start();
            player.play();
            recordBtn.innerText = "ENREGISTREMENT EN COURS...";
            recordBtn.style.background = "#ff3e3e";
        } else {
            mediaRecorder.stop();
            player.pause();
        }
    };

    // Arrêt automatique à la fin
    player.onended = () => {
        if (mediaRecorder && mediaRecorder.state === "recording") {
            mediaRecorder.stop();
        }
    };
</script>
